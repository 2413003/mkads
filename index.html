🗄️ Database Schema
Run this SQL in your Supabase SQL editor to create the required tables:
-- Campaigns
CREATE TABLE campaigns (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  name TEXT NOT NULL,
  objective TEXT,
  status TEXT DEFAULT 'draft',
  budget NUMERIC DEFAULT 0,
  total_budget NUMERIC DEFAULT 0,
  bidding_strategy TEXT DEFAULT 'maximise_conversions',
  billing_model TEXT DEFAULT 'prepaid',
  commission_rate NUMERIC,
  commission_trigger TEXT DEFAULT 'completed_sale',
  sales_process_owner TEXT DEFAULT 'advertiser',
  spent NUMERIC DEFAULT 0,
  impressions BIGINT DEFAULT 0,
  clicks BIGINT DEFAULT 0,
  conversions INT DEFAULT 0,
  placements TEXT[] DEFAULT '{}',
  start_date DATE,
  end_date DATE,
  created_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE campaigns ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users own campaigns" ON campaigns
  FOR ALL USING (auth.uid() = user_id);

-- Audiences
CREATE TABLE audiences (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  name TEXT NOT NULL,
  age_min INT DEFAULT 18,
  age_max INT DEFAULT 65,
  locations TEXT[] DEFAULT '{}',
  interests TEXT[] DEFAULT '{}',
  size_estimate BIGINT DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE audiences ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users own audiences" ON audiences
  FOR ALL USING (auth.uid() = user_id);

-- Analytics daily
CREATE TABLE analytics_daily (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  campaign_id UUID REFERENCES campaigns(id) ON DELETE CASCADE,
  date DATE NOT NULL,
  impressions BIGINT DEFAULT 0,
  clicks BIGINT DEFAULT 0,
  spend NUMERIC DEFAULT 0,
  conversions INT DEFAULT 0
);
